with(Domains):

Test:=proc(f,q,k)
local deg_f, p, R, h, i, g, M, F;

p := q^k;
deg_f := degree(f);

R := DUP(Zmod(p), x);
F := R[Input](f);
h := R[Rem](x, F);

i := 1;
while i <= Z[Quo](deg_f, 2) do
    h := R[Rem](h^p, F);
    g := R[Input](h - x);
    if R[Gcd](g, F) <> R[1] then
        return "Reducible";
    end if;
    i := i + 1;
od:


return("Irreducible")  
end:

#EXAMPLES:

Test(x^4 + x^3 + x^2 + x + 1, 3, 1); #Result: "Irreducible"

Test(x^2 + x + 1, 3, 1); #Result: "Reducible"
